'''-----------------------------Data Cleaning and Preprocessing pipeline----------------------------------'''

#Removing special character
def remove_special_character(content):
    return re.sub('\W+',' ', content )#re.sub('\[[^&@#!]]*\]', '', content)

# Removing URL's
def remove_url(content):
    return re.sub(r'http\S+', '', content)

#Removing the stopwords from text
def remove_stopwords(content):
    clean_data = []
    for i in content.split():
        if i.strip().lower() not in stop_words and i.strip().lower().isalpha():
            clean_data.append(i.strip().lower())
    return " ".join(clean_data)

# Expansion of english contractions
def contraction_expansion(content):
    content = re.sub(r"won\'t", "would not", content)
    content = re.sub(r"can\'t", "can not", content)
    content = re.sub(r"don\'t", "do not", content)
    content = re.sub(r"shouldn\'t", "should not", content)
    content = re.sub(r"needn\'t", "need not", content)
    content = re.sub(r"hasn\'t", "has not", content)
    content = re.sub(r"haven\'t", "have not", content)
    content = re.sub(r"weren\'t", "were not", content)
    content = re.sub(r"mightn\'t", "might not", content)
    content = re.sub(r"didn\'t", "did not", content)
    content = re.sub(r"n\'t", " not", content)
    '''content = re.sub(r"\'re", " are", content)
    content = re.sub(r"\'s", " is", content)
    content = re.sub(r"\'d", " would", content)
    content = re.sub(r"\'ll", " will", content)
    content = re.sub(r"\'t", " not", content)
    content = re.sub(r"\'ve", " have", content)
    content = re.sub(r"\'m", " am", content)'''
    return content

#Data preprocessing
def data_cleaning(content):
    content = str(content)
    content = contraction_expansion(content)
    content = remove_special_character(content)
    content = remove_url(content)
    
    content = remove_stopwords(content)    
    return content

def get_analysis_text(row):
    """Priority: review > feedback > comments > pros+cons > summary"""
    
    if 'review' in df.columns and pd.notna(row.get('review')):
        return str(row['review'])
    
    if 'feedback' in df.columns and pd.notna(row.get('feedback')):
        return str(row['feedback'])
    
    if 'comments' in df.columns and pd.notna(row.get('comments')):
        return str(row['comments'])
    
    #  Combined pros/cons (no labels)
    pros_cons = []
    if pd.notna(row.get('pros')):
        pros_cons.append(str(row['pros']))
    if pd.notna(row.get('cons')):
        pros_cons.append(str(row['cons']))
    if pros_cons:
        return ' '.join(pros_cons).strip()

   #  Summary data
    if 'summary' in df.columns and pd.notna(row.get('summary')):
        return str(row['summary'])
    
    return ""

from sklearn.feature_selection import chi2
import numpy as np
from prettytable import PrettyTable
import seaborn as sns
import matplotlib.pyplot as plt

# Use TF-IDF or CountVectorizer features
X = x_train_tfidf 
y = y_train
classes = np.unique(y)

# Get feature names
feature_names = tfidfvect.get_feature_names_out() 

# Collect top features per class
top_N = 20  # Top N features per class
top_features_per_class = {}

for cls in classes:
    binary_y = (y == cls).astype(int)
    chi2_scores, _ = chi2(X, binary_y)
    sorted_indices = np.argsort(chi2_scores)[::-1][:top_N]
    top_features = [feature_names[i] for i in sorted_indices]
    top_scores = [chi2_scores[i] for i in sorted_indices]
    top_features_per_class[cls] = (top_features, top_scores)
all_top_features = sorted(set(f for features, _ in top_features_per_class.values() for f in features))

heatmap_data = []

for feature in all_top_features:
    row = []
    for cls in classes:
        features, scores = top_features_per_class[cls]
        if feature in features:
            row.append(scores[features.index(feature)])
        else:
            row.append(0.0)
    heatmap_data.append(row)
heatmap_array = np.array(heatmap_data)
plt.figure(figsize=(12, max(6, len(all_top_features) * 0.3)))
sns.heatmap(
    heatmap_array,
    xticklabels=[f"Class {cls}" for cls in classes],
    yticklabels=all_top_features,
    cmap="YlGnBu",
    annot=True,
    fmt=".1f",
    linewidths=0.5,
    cbar=True
)
plt.title(f"Top {top_N} Chi2 Features per Class")
plt.xlabel("Class")
plt.ylabel("Feature")
plt.tight_layout()
plt.show()
